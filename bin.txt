const str1 = 'Hello World';
const str2 = 'Heeyy World';
const findHammingDistance = (str1 = '', str2 = '') => {
   	let distance = 0;
   	if(str1.length === str2.length) {
      	for (let i = 0; i < str1.length; i++) {
         	if (str1[i].toLowerCase() != str2[i].toLowerCase()){
            	distance++
         	}
      	}
      	return distance
   	};
   	return 0;
};
